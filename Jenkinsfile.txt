pipeline {
    agent any

    environment {
        DOCKER_CREDS = credentials('docker-hub')
    }

    stages {
        stage('Clone Repo') {
            steps {
                git branch: 'main', url: 'https://github.com/Meghana2417/InventoryService.git'
            }
        }

        stage('Docker Build') {
            steps {
                sh "docker build --no-cache -t meghana1724/inventoryservice:latest ."
            }
        }

        stage('Docker Login') {
            steps {
                sh 'echo "$DOCKER_CREDS_PSW" | docker login -u "$DOCKER_CREDS_USR" --password-stdin'
            }
        }

        stage('Docker Push') {
            steps {
                sh "docker push meghana1724/inventoryservice:latest"
            }
        }

        stage('Deploy') {
            steps {
                sh '''
                docker stop inventoryservice || true
                docker rm inventoryservice || true

                docker pull meghana1724/inventoryservice:latest

                docker run -d --name api -p 8004:8004 meghana1724/inventoryservice:latest
                '''
            }
        }
    }
}
